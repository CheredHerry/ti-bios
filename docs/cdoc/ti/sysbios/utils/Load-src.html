    51    /*!
    52     *  ======== Load ========
    53     *  The Load module reports execution times and load information for threads 
    54     *  in a system. 
    55     *
    56     *  SYS/BIOS manages four distinct levels of execution threads: hardware
    57     *  interrupt service routines, software interrupt routines, tasks, and
    58     *  background idle functions. This module reports execution time and load on a 
    59     *  per-task basis, and also provides information globally for hardware 
    60     *  interrupt service routines, software interrupt routines and idle functions 
    61     *  (in the form of the idle task). It can also report an estimate of the 
    62     *  global CPU load, which is computed as the percentage of time in the
    63     *  benchmark window which was NOT spent in the idle loop. 
    64     *  More specifically, the load is computed as follows:
    65     *
    66     *  global CPU load = 100 * (1 - (min. time for a trip around idle loop * 
    67     *      # times in idle loop)/(benchmark time window) )
    68     *      
    69     *  Any work done in the idle loop is included in the CPU load - in other 
    70     *  words, any time spent in the loop beyond the shortest trip around the idle
    71     *  loop is counted as non-idle time.
    72     * 
    73     *  Execution time is reported in units of 
    74     *  {@link xdc.runtime.Timestamp Timestamp} counts, and load is reported 
    75     *  in percentages.
    76     *
    77     *  By default, load data is gathered in all threads. 
    78     *  {@link #hwiEnabled hwiEnabled}, {@link #swiEnabled swiEnabled} 
    79     *  and {@link #taskEnabled taskEnabled} can be used to select which 
    80     *  type(s) of threads are monitored. These statistics are automatically 
    81     *  recorded to the load module's logger instance. Users can also choose 
    82     *  to call {@link #getTaskLoad}, {@link #getGlobalSwiLoad}, 
    83     *  {@link #getGlobalHwiLoad} and {@link #getCPULoad} at any time to obtain the
    84     *  statistics at runtime.   
    85     *  
    86     *  The module relies on {@link #update} to be called to compute load and 
    87     *  execution times from the time when {@link #update} was last called. This
    88     *  is automatically done for every period specified by 
    89     *  {@link #windowInMs windowInMs} in an {@link ti.sysbios.knl.Idle Idle} 
    90     *  function when {@link #updateInIdle updateInIdle} is 
    91     *  set to true. The time between 2 calls to {@link #update} is called the 
    92     *  benchmark time window. 
    93     *     
    94     *  By passing in a function pointer of type 
    95     *  Void(*{@link #postUpdate})(Void) through the 
    96     *  {@link #postUpdate postUpdate} config parameter, one can specify a 
    97     *  {@link #postUpdate postUpdate} function that is 
    98     *  automatically called by {@link #update} immediately after the statistics 
    99     *  have been computed at the end of a benchmark time window.  Setting this 
   100     *  function allows the user to optionally post-process the thread statistics 
   101     *  once and only once per benchmark window. 
   102     * 
   103     *  Advanced users could optionally omit calling {@link #update} in the idle 
   104     *  loop by setting {@link #updateInIdle updateInIdle} to false. They can 
   105     *  then call {@link #update} somewhere else (e.g. in a periodic higher 
   106     *  priority Task) instead to ensure statistics are computed even when the 
   107     *  system is never idle.  
   108     *     
   109     *  @a(Examples)
   110     *  Configuration example: The following statements configure the Load module
   111     *  to obtain output in the CCS output window. This is useful for early
   112     *  development without RTA tools support in SYS/BIOS:
   113     *
   114     *  @p(code)
   115     *  var System = xdc.useModule('xdc.runtime.System');
   116     *  var SysStd = xdc.useModule('xdc.runtime.SysStd');
   117     *  System.SupportProxy = SysStd;
   118     *
   119     *  var Load = xdc.useModule('ti.sysbios.utils.Load');
   120     *  var LoggerSys = xdc.useModule('xdc.runtime.LoggerSys');
   121     *  var Defaults = xdc.useModule('xdc.runtime.Defaults');
   122     *  var Diags = xdc.useModule('xdc.runtime.Diags');
   123     *
   124     *  // Turn on statistics log messages for Load module
   125     *  Load.common$.diags_USER4 = Diags.ALWAYS_ON;
   126     *  // Define a program-wide logger to be the System Logger
   127     *  Defaults.common$.logger = LoggerSys.create();
   128     *  @p
   129     *
   130     *  Notes on above example:
   131     *  @p(blist)
   132     *  - When using the {@link xdc.runtime.LoggerSys System Logger},
   133     *  {@link xdc.runtime.System#printf System_printf} calls are made to produce
   134     *  the output in the output window. These calls are highly intrusive
   135     *  and are only suited for early development.
   136     *
   137     *  - Also make sure your task stack sizes are large enough when using the
   138     *  {@link xdc.runtime.LoggerSys System Logger} - particularly the idle task
   139     *  stack size since logging is done in the idle task by default.
   140     *  @p 
   141     * 
   142     *  @a(Caveats)
   143     *  @p(nlist)
   144     *  - For the module to return accurate load values, the {@link #update} 
   145     *  function must be run at least once before the Timestamp count gets to 
   146     *  wrap around. This means on a platform with a 32-bit 
   147     *  {@link xdc.runtime.Timestamp Timestamp} frequency that runs at 200 MHz, 
   148     *  the function must be called at least once every 21 sec. With faster 
   149     *  {@link xdc.runtime.Timestamp Timestamp} frequencies it would have to be 
   150     *  called even more frequently. 
   151     *   
   152     *  - If the {@link #updateInIdle updateInIdle} option is enabled, the 
   153     *  module will call {@link #update} periodically according to 
   154     *  {@link #windowInMs windowInMs}. The latter must be set to a time 
   155     *  interval that is below 
   156     *  (2^32) * 1000 / ({@link xdc.runtime.Timestamp Timestamp} Frequency) 
   157     *  milliseconds
   158     *
   159     *  - If the {@link #updateInIdle updateInIdle} option is disabled, the user is
   160     *  then responsible for calling {@link #update} more often than the above 
   161     *  time interval.
   162     *   
   163     *  - Load values might not add up to 100%. Because the module minimizes 
   164     *  its interrupt latency and avoids locking up the system to compute all
   165     *  thread loads at once, each thread's execution time is evaluated separately
   166     *  over its own time window, which could be slightly different to that of
   167     *  another thread.
   168     *   
   169     *  - When disabling monitoring of a thread type of higher priority, the 
   170     *  time spent in that thread type will be counted towards time spent in 
   171     *  threads of lower priority. E.g. if {@link #hwiEnabled hwiEnabled} and 
   172     *  {@link #swiEnabled swiEnabled} are both false, then time spent in Hwi's and
   173     *  Swi's will be counted towards the Tasks in which the interrupts happened. 
   174     *  Thus, for better accuracy, it is best to leave monitoring on for threads of
   175     *  a higher priority relative to the thread type of interest.  
   176     *   
   177     *  - The implementation of {@link #getCPULoad()} self-calibrates the shortest 
   178     *  path through the idle loop. It does this by keeping track of the shortest 
   179     *  time between invocations of an idle function automatically inserted by the 
   180     *  Load module, and assumes that to be the time it takes for one iteration
   181     *  through the idle loop. Because of this, the CPU load value is only an 
   182     *  estimate since the idle loop might occasionally take longer to run 
   183     *  (e.g. due to caching effects, stalls). The reported CPU load tends to be 
   184     *  slightly higher than reality, especially when the load is low. 
   185     *   
   186     *  - If {@link #taskEnabled taskEnabled} is set to true, 
   187     *  task name support will be automatically turned on so that the Load module 
   188     *  can print out the task names in its log output.
   189     *  
   190     *  - Currently does not support {@link xdc.runtime.Timestamp Timestamp} 
   191     *  frequencies over 4 GHz.       
   192     *  @p
   193     *  
   194     *  @p(html)
   195     *  <h3> Calling Context </h3>
   196     *  <table border="1" cellpadding="3">
   197     *    <colgroup span="1"></colgroup> <colgroup span="5" align="center">
   198     *    </colgroup>
   199     *
   200     *    <tr><th> Function                 </th><th>  Hwi   </th><th>  Swi   </th>
   201     *    <th>  Task  </th><th>  Main  </th><th>  Startup  </th></tr>
   202     *    <!--                                                          -->
   203     *    <tr><td> {@link #getCPULoad}      </td><td>   Y    </td><td>   Y    </td>
   204     *    <td>   Y    </td><td>   N    </td><td>   N    </td></tr>
   205     *    <tr><td> {@link #getGlobalHwiLoad}</td><td>   Y    </td><td>   Y    </td>
   206     *    <td>   Y    </td><td>   N    </td><td>   N    </td></tr>
   207     *    <tr><td> {@link #getGlobalSwiLoad}</td><td>   Y    </td><td>   Y    </td>
   208     *    <td>   Y    </td><td>   N    </td><td>   N    </td></tr>
   209     *    <tr><td> {@link #getTaskLoad}     </td><td>   Y    </td><td>   Y    </td>
   210     *    <td>   Y    </td><td>   N    </td><td>   N    </td></tr>
   211     *    <tr><td> {@link #reset}           </td><td>   Y*   </td><td>   Y*   </td>
   212     *    <td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   213     *    <tr><td> {@link #update}          </td><td>   Y*   </td><td>   Y*   </td>
   214     *    <td>   Y    </td><td>   N    </td><td>   N    </td></tr>
   215     *    <tr><td colspan="6"> Definitions: <br />
   216     *       <ul>
   217     *         <li> <b>Hwi</b>: API is callable from a Hwi thread. </li>
   218     *         <li> <b>Swi</b>: API is callable from a Swi thread. </li>
   219     *         <li> <b>Task</b>: API is callable from a Task thread. </li>
   220     *         <li> <b>Main</b>: API is callable during any of these phases: </li>
   221     *           <ul>
   222     *             <li> In your module startup after this module is started 
   223     *    (e.g. Load_Module_startupDone() returns TRUE). </li>
   224     *             <li> During xdc.runtime.Startup.lastFxns. </li>
   225     *             <li> During main().</li>
   226     *             <li> During BIOS.startupFxns.</li>
   227     *           </ul>
   228     *         <li> <b>Startup</b>: API is callable during any of these phases:</li>
   229     *           <ul>
   230     *             <li> During xdc.runtime.Startup.firstFxns.</li>
   231     *             <li> In your module startup before this module is started 
   232     *    (e.g. Load_Module_startupDone() returns FALSE).</li>
   233     *           </ul>
   234     *       <li> <b>*</b>: Indicates only when {@link #taskEnabled taskEnabled} 
   235     *    is set to false. </li>
   236     *       </ul>
   237     *    </td></tr>
   238     *  </table>
   239     *  @p  
   240     */

   368        /*!
   369         *  ======== updateInIdle ========
   370         *  Automatically call {@link #update} in Idle loop
   371         *
   372         *  If this parameter is set to `true`, an {@link ti.sysbios.knl.Idle Idle}
   373         *  function, which computes and updates the CPU load, is added to the
   374         *  {@link ti.sysbios.knl.Idle#run Idle loop}.
   375         */     
   376        config Bool updateInIdle = true;
   377    
   378        /*!
   379         *  ======== minIdle ========
   380         *  Specifies the minimum time used to compute idle time
   381         *
   382         *  During CPU load measurement, the minimum time observed executing
   383         *  the idle loop together with the count of the number of times the idle
   384         *  loop runs id used to compute the total idle time.  However, since the
   385         *  _actual_ time spent in the idle loop will almost always be greater
   386         *  then the observed minimum, the idle time is almost always under
   387         *  estimated.
   388         *
   389         *  This configuration parameter can be used to compensate for this
   390         *  underestimate by placing a lower bound on the minimum time used in
   391         *  the CPU measurements.  The minimum idle time used to compute CPU
   392         *  load is guarenteed to be greater than or equal to `minIdle`.
   393         */
   394        metaonly config UInt32 minIdle = 0;
   395        
   396        /*!
   397         *  ======== windowInMs ========
   398         *  Load averaging period
   399         *
   400         *  When {@link #updateInIdle} is enabled, this period determines the
   401         *  minimum elapsed time between updates of the load statistics in the
   402         *  idle loop.
   403         */
   404        config UInt windowInMs = 500;

   413         *  {@link #LS_hwiLoad event} with this cumulative time is logged.  This
   414         *  time, together with the total elapsed time, is used to estimate the
   415         *  CPU load due to any Hwi instances run between these updates.
   416         */
   417        config Bool hwiEnabled = false;
   418    
   419        /*!
   420         *  ======== swiEnabled ========

   423         *  When this parameter is set to `true`, Swi processing time is

   425         *  part of Load_update, an {@link #LS_swiLoad event} with this
   426         *  cumulative time is logged.  This time together with the total
   427         *  elapsed time is used to estimate the CPU load due to any Swi
   428         *  instances run between these updates.
   429         */
   430        config Bool swiEnabled = false;
   431

   433         *  ======== taskEnabled ========

   439         *  cumulative time is logged for each task.  This time together with the
   440         *  total elapsed time is used to estimate the CPU load due to each Task
   441         *  run between these updates.
   442         */
   443        config Bool taskEnabled = true;
